query Invoices($startDate: DateTime, $endDate: DateTime, $childId: ID, $status: [InvoiceStatus], $paymentMode: PaymentMode) {
  invoices(
    startDate: $startDate
    endDate: $endDate
    childId: $childId
    status: $status
    paymentMode: $paymentMode
  ) {
    ...InvoiceParts
    __typename
  }
}
